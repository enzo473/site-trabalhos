  <script>
    const templates = document.querySelectorAll('.template-card');
    templates.forEach(t => t.addEventListener('click', ()=>{
      templates.forEach(x=>x.classList.remove('selected'));
      t.classList.add('selected');
    }));

    function gerarNivelTexto(nivel){
      if(nivel === "Ensino Fundamental") return "Explicação simples, com frases curtas e exemplos fáceis de entender.";
      if(nivel === "Ensino Médio") return "Explicação intermediária, com conceitos claros e linguagem objetiva.";
      return "Explicação aprofundada, com conceitos técnicos e estrutura formal superior.";
    }

    function localGenerator(p){
      const nivelTexto = gerarNivelTexto(p.nivel);
      const depth = p.iaNivel === 'forte' ? 3 : p.iaNivel === 'media' ? 2 : 1;

      return `
        <h1 style='text-align:center'>${p.tema}</h1>
        <h3>Introdução</h3>
        <p>${nivelTexto} (Nível IA: ${depth})</p>

        <h3>Desenvolvimento</h3>
        <p>Conteúdo desenvolvido sobre ${p.tema}, adaptado ao nível escolhido.</p>

        <h3>Conclusão</h3>
        <p>Resumo final com fechamento apropriado ao tema.</p>
      `;
    }

    document.getElementById('gerarBtn').addEventListener('click', ()=>{
      const p = {
        tema: document.getElementById('tema').value.trim(),
        nivel: document.getElementById('nivel').value,
        iaNivel: document.getElementById('iaNivel').value,
      };

      document.getElementById('preview').innerHTML = localGenerator(p);
    });
  </script>
